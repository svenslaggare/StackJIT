func println(Ref.Array[Char]) Void
{
	.locals 1
	.local 0 Int
	
	PUSHINT 0
	STLOC 0

	LDLOC 0
	LDARG 0
	LDLEN
	BGE 15

	LDARG 0
	LDLOC 0
	LDELEM Char
	CALL rt.printchar(Char)

	LDLOC 0
	PUSHINT 1
	ADD
	STLOC 0

	BR 2
	
	PUSHCHAR 10
	CALL rt.printchar(Char)

	RET
}

func main() Int
{
	LDSTR "Hello, World!"
	CALL println(Ref.Array[Char])

	PUSHINT 0
	RET
}
