func createCharArray() Ref.Array[Char]
{
	.locals 1
	.local 0 Ref.Array[Char]

	LDINT 4
	NEWARR Char
	STLOC 0

	LDLOC 0
	LDINT 0
	LDCHAR 65
	STELEM Char

	LDLOC 0
	LDINT 1
	LDCHAR 66
	STELEM Char

	LDLOC 0
	LDINT 2
	LDCHAR 67
	STELEM Char

	LDLOC 0
	LDINT 3
	LDCHAR 68
	STELEM Char

	LDLOC 0
	RET
}

func main() Int
{
	.locals 2
	.local 0 Ref.Array[Char]
	.local 1 Ref.Class.std.String

	CALL createCharArray()
	STLOC 0

	LDLOC 0
	NEWOBJ std.String::.constructor(Ref.Array[Char])
	STLOC 1

	LDLOC 0
	LDINT 0
	LDCHAR 75
	STELEM Char

	LDLOC 0
	CALL std.println(Ref.Array[Char])

	LDLOC 1
	CALL std.println(Ref.Class.std.String)

	LDINT 0
	RET
}
