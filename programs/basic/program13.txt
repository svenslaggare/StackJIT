func addInt(Int Int) Int
{
	LDARG 0
	LDARG 1
	ADD
	RET
}

func println(Bool) Void
{
   LDARG 0
   PUSHTRUE
   BNE 6
   LDINT 1
   CALL std.println(Int)
   BR 8
   LDINT 0
   CALL std.println(Int)
   RET
}

func main() Int
{
	LDINT 4711
	CALL std.println(Int)

	LDFLOAT 13.37
	CALL std.println(Float)

	PUSHTRUE
	CALL println(Bool)

	LDINT 0
	RET
}
