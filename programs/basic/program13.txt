func addInt(Int Int) Int
{
	LDARG 0
	LDARG 1
	ADD
	RET
}

func println(Bool) Void
{
   LDARG 0
   PUSHTRUE
   BNE 6
   PUSHINT 1
   CALL std.println(Int)
   BR 8
   PUSHINT 0
   CALL std.println(Int)
   RET
}

func main() Int
{
	PUSHINT 4711
	CALL std.println(Int)

	PUSHFLOAT 13.37
	CALL std.println(Float)

	PUSHTRUE
	CALL println(Bool)

	PUSHINT 0
	RET
}
