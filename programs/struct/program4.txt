struct Point
{
	x Float
	y Float
}

member Point::.constructor() Void
{
   RET
}

func Point::length(Ref.Struct.Point) Float
{
	.locals 1
	LDARG 0
	STLOC 0

	LDLOC 0
	LDFIELD Point::x
	LDLOC 0
	LDFIELD Point::x
	MUL

	LDLOC 0
	LDFIELD Point::y
	LDLOC 0
	LDFIELD Point::y
	MUL

	ADD
	CALL sqrt(Float)

	RET
}

func main() Int
{
	.locals 1
	
	NEWOBJ Point::.constructor()
	STLOC 0

	LDLOC 0
	PUSHFLOAT 1.23
	STFIELD Point::x

	LDLOC 0
	PUSHFLOAT 2.23
	STFIELD Point::y

	LDLOC 0
	CALL Point::length(Ref.Struct.Point)
	CALL rt.println(Float)

	PUSHINT 0
	RET
}
