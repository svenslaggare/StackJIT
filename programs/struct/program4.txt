struct Point
{
	x Int
	y Int
}

member Point::.constructor() Void
{
   RET
}

func printPoint(Ref.Struct.Point) Void
{
	LDARG 0
	LDFIELD Point::x
	CALL std.print(Int)

	LDCHAR 58
    CALL std.printchar(Char)

	LDARG 0
	LDFIELD Point::y
	CALL std.print(Int)

	LDCHAR 10
	CALL std.printchar(Char)

	RET
}

func main() Int
{
	.locals 1

	NEWOBJ Point::.constructor()
	STLOC 0

	LDLOC 0
	LDINT 1337
	STFIELD Point::x

	LDLOC 0
	LDINT 4711
	STFIELD Point::y

	LDLOC 0
	CALL printPoint(Ref.Struct.Point)

	LDINT 0
	RET
}
