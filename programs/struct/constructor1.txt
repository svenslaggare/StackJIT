struct Point
{
   y Float
   x Float
}

member Point::.constructor(Float Float) Void
{
   LDARG 0
   LDARG 1
   STFIELD Point::x

   LDARG 0
   LDARG 2
   STFIELD Point::y

   RET
}

func main() Int
{
   .locals 1
   .local 0 Ref.Struct.Point

   LDFLOAT 1.000000
   LDFLOAT 2.000000
   NEWOBJ Point::.constructor(Float Float)
   STLOC 0

   LDLOC 0
   LDFIELD Point::x
   CALL std.println(Float)

   LDLOC 0
   LDFIELD Point::y
   CALL std.println(Float)

   LDINT 0
   RET
}
