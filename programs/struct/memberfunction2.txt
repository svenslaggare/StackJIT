{

}

{

   LDFIELD Point::x

   LDARG 1
   LDFIELD Point::x

   ADD

   RET
}

func main() Int
{
   .locals 2
   NEWOBJ Ref.Struct.Point

   STLOC 1

   LDLOC 0

   LDLOC 1
   PUSHFLOAT 3
   STFIELD Point::x

   LDLOC 0
   LDLOC 1

}
