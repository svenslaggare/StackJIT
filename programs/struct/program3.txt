struct Point
{
	x Int
	y Int
}

struct Point3D
{
	x Int
	y Int
	z Int
}

func main() Int
{
	.locals 0

	#PUSHINTNULL

	#PUSHINT 2
	#NEWARR Int

	NEWOBJ Ref.Struct.Point3D
	#NEWOBJ Ref.Struct.Point

	LDFIELD Point3D::z

	#PUSHINT 0
	#STFIELD Point::x
	#PUSHINT 0

	#PUSHINT 0
	#LDELEM Int

	#PUSHINT 0
	#PUSHINT 0
	#STELEM Int
	#PUSHINT 0

	RET
}
