class Point
{
   y Float
   x Float
}

member Point::.constructor() Void
{
   RET
}

member Point::addX(Ref.Class.Point) Float
{
   LDARG 0
   LDFIELD Point::x

   LDARG 1
   LDFIELD Point::x

   ADD

   RET
}

func main() Int
{
   .locals 2
   NEWOBJ Point::.constructor()
   STLOC 0

   NEWOBJ Point::.constructor()
   STLOC 1

   LDLOC 0
   LDFLOAT 2
   STFIELD Point::x

   LDLOC 1
   LDFLOAT 3
   STFIELD Point::x

   LDLOC 0
   LDLOC 1
   CALLINST Point::addX(Ref.Class.Point)
   CALL std.println(Float)

   LDINT 0
   RET
}
