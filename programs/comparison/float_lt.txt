func println(Bool) Void
{
   LDARG 0
   PUSHTRUE
   BNE 6
   LDINT 1
   CALL std.println(Int)
   BR 8
   LDINT 0
   CALL std.println(Int)
   RET
}

func main() Int
{
   LDFLOAT 3.000000
   LDFLOAT 4.000000
   CMPLT
   CALL println(Bool)

   LDFLOAT 4.000000
   LDFLOAT 4.000000
   CMPLT
   CALL println(Bool)

   LDFLOAT 5.000000
   LDFLOAT 4.000000
   CMPLT
   CALL println(Bool)

   LDINT 4711
   RET
}
