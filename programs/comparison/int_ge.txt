func println(Bool) Void
{
   LDARG 0
   LDTRUE
   BNE 6
   LDINT 1
   CALL std.println(Int)
   BR 8
   LDINT 0
   CALL std.println(Int)
   RET
}

func main() Int
{
   LDINT 30
   LDINT 40
   CMPGE
   CALL println(Bool)

   LDINT 40
   LDINT 40
   CMPGE
   CALL println(Bool)

   LDINT 50
   LDINT 40
   CMPGE
   CALL println(Bool)

   LDINT 4711
   RET
}
